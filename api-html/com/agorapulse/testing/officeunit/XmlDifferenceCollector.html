18 package com.agorapulse.testing.officeunit;
19 
20 import org.xmlunit.builder.DiffBuilder;
21 import org.xmlunit.builder.Input;
22 import org.xmlunit.diff.ComparisonResult;
23 import org.xmlunit.diff.Diff;
24 import org.xmlunit.diff.Difference;
25 import org.xmlunit.diff.DifferenceEvaluators;
26 
27 import javax.xml.transform.Source;
28 import java.io.InputStream;
29 import java.util.ArrayList;
30 import java.util.List;
31 import java.util.Set;
32 import java.util.stream.Collectors;
33 
34 public enum XmlDifferenceCollector implements DifferenceCollector {
35 
36     INSTANCE;
37 
38     public List<DocumentDifference> computeDifferences(String path, InputStream actualStream, InputStream expectedStream, Set<String> ignoredPaths) {
39         List<Difference> ret = new ArrayList<>();
40         Source input = Input.fromStream(actualStream).build();
41         Source output = Input.fromStream(expectedStream).build();
42 
43         Diff diff = DiffBuilder
44                 .compare(input)
45                 .withTest(output)
46                 .ignoreComments()
47                 .ignoreWhitespace()
48                 .checkForSimilar()
49                 .withDifferenceEvaluator(DifferenceEvaluators.chain(
50                         DifferenceEvaluators.Default,
51                         (comparison, outcome) -> {
52                             if (ComparisonResult.DIFFERENT.equals(outcome)) {
53                                 for (String ignoredPath : ignoredPaths) {
54                                     if (comparison.getControlDetails().getXPath() != null && comparison.getControlDetails().getXPath().endsWith(ignoredPath)) {
55                                         return ComparisonResult.EQUAL;
56                                     }
57                                 }
58                             }
59                             return outcome;
60                         }
61                 ))
62                 .build();
63 
64         for (Difference difference : diff.getDifferences()) {
65             ret.add(difference);
66         }
67 
68         return ret.stream().map(difference -> new XmlDocumentDifference(path, difference)).collect(Collectors.toList());
69     }
70 }
